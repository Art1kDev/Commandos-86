# Commands
help - show help panel;

dir - check directory;

cls - clear all message;

ver - show system version;

echo (Text) - print your text;

reboot - restart system;

date - show date;

shutdown - power off system

Hworld - hello world

mem - check memory

mkdir X - make new folder in directory

col X - (0-5) change bg color
